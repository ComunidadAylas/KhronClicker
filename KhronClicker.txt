Shinies
	*Pinguino
		on click:log Woop
		movement:onLeft moveRight fade bounce:0.01
		frequency:60
		frequency variation:120
		icon:https://raw.githubusercontent.com/ComunidadAylas/KhronClicker/master/Imagenes/pinguino_adorable_256.png
		duration: 5
 
		class:bigButton
		on click:
			$boost=1
			if (have PinguinoMejora3) $boost=2
			if (chance(90%))
				//Al menos 1, o entre 1 y 2 minutos de producción de Khrones
				$amount=max(1,random(khrones:ps*60*1,khrones:ps*60*2))*$boost
				toast ¡El <b><#0094FF>pingüino</#></b> te bendice con<//><b>[$amount] Ꝃhrones</b>!
				yield $amount khrones
				yield 1 PinguinoClic
			else
				$amount=1
				toast ¡El <b><#0094FF>pingüino</#></b> ha atraído a <//><b>[$amount] Donante[s?$amount]</b> de Khron!
				yield 1 Donante
				yield 1 PinguinoClic
			end
		end

	*Loro
		on click:log Woop
		movement:onRight moveLeft fade bounce:0.05
		frequency:240
		frequency variation:180
		icon:https://raw.githubusercontent.com/ComunidadAylas/KhronClicker/master/Imagenes/loro_adorable_256.png
		class:bigButton
		on click:
			$boost=1
			if (have LoroMejora3) $boost=3
			if (chance(10%))
				$amount=1
				toast ¡El <b><#FF0000>loro</#></b> ha atraído a <//><b>[$amount] Donante[s?$amount]</b> de Khron!
				yield 1 Donante
				yield 1 LoroClic
			else
				$amount=7*$boost
				toast ¡El <b><#FF0000>loro</#></b> te bendice con<//><b>[$amount] Esmeralda[s?$amount]</b> de Khron!
				yield $amount EsmeraldaKhron
				yield 1 LoroClic
			end
		end

	*Corazon
		on click:log Woop
		movement:anywhere pulse fade
		frequency:420
		frequency variation:180
		icon:https://raw.githubusercontent.com/ComunidadAylas/KhronClicker/master/Imagenes/corazon_arcoiris_256.png
		duration: 15
		class:bigButton
		on click:
			$boost=1
			if (have CorazonMejora3) $boost=4
			if (chance(20%))
				//Al menos 5, o entre 5 y 10 minutos de producción de Khrones
				$amount=max(5,random(khrones:ps*60*5,khrones:ps*60*10))*$boost
				toast ¡El <b><#BA212A>c</#><#E76935>o</#><#FBD52B>r</#><#74B833>a</#><#74B833>z</#><#2B62B0>o</#><#6B3293>n</#></b> te bendice con <//><b>[$amount] Ꝃhrones</b>!
				yield $amount khrones
				yield 1 CorazonClic
			else
				$amount=1*$boost
				toast ¡El <b><#BA212A>c</#><#E76935>o</#><#FBD52B>r</#><#74B833>a</#><#74B833>z</#><#2B62B0>o</#><#6B3293>n</#></b> te bendice con<//><b>[$amount] Donante[s?$amount]</b> de Khron!
