Shinies
	*Pinguino
		on click:log Woop
		movement:onLeft moveRight fade bounce:0.01
		frequency:60
		frequency variation:120
		icon:https://cdn.discordapp.com/attachments/431492337290575882/1111285186303365130/pinguino_adorable.png
		duration: 5
 
		class:bigButton
		on click:
			$boost=1
			if (have PinguinoMejora3) $boost=2
			if (chance(90%))
				//Al menos 5, o entre 1 y 3 minutos de producción de Khrones
				$amount=max(5,random(khrones:ps*60*1,khrones:ps*60*3))*$boost
				toast El Pingüinito te bendice con<//><b>[$amount] Ꝃhrones</b>.
				yield $amount khrones
			else
				$amount=1
				toast ¡El Pingüinito ha atraído a <//><b>[$amount] Donante[s?$amount]</b> de Khron!
				yield 1 Donante
			end
		end

	*Loro
		on click:log Woop
		movement:onRight moveLeft fade bounce:0.05
		frequency:240
		frequency variation:180
		icon:https://cdn.discordapp.com/attachments/1109612171572285470/1110989203057164288/loro_adorable256x256.png
		class:bigButton
		on click:
			$boost=1
			if (have LoroMejora3) $boost=3
			if (chance(10%))
				//Al menos 10, o entre 15 y 30 minutos de producción de Khrones
				$amount=max(10,random(khrones:ps*60*15,khrones:ps*60*30))*$boost
				toast ¡El Loro te bendice con<//><b>[$amount] Ꝃhrones</b>!
				yield $amount khrones
			else
				$amount=5*$boost
				toast ¡El Loro te bendice con<//><b>[$amount] Esmeralda[s?$amount]</b> de Khron!
				yield $amount EsmeraldaKhron
			end
		end

	*Corazon
		on click:log Woop
		movement:anywhere pulse fade
		frequency:420
		frequency variation:180
		icon:https://cdn.discordapp.com/attachments/431492337290575882/1111618573681561660/corazon_arcoiris_256.png
		duration: 30
		class:bigButton
		on click:
			$boost=1
			if (have PinguinoMejora3) $boost=4
			if (chance(50%))
				//Al menos 15, o entre 30 y 60 minutos de producción de Khrones
				$amount=max(15,random(khrones:ps*60*30,khrones:ps*60*60))*$boost
				toast El Corasoncito te bendice con <//><b>[$amount] Ꝃhrones</b>.
				yield $amount khrones
			else
				$amount=1*$boost
				toast ¡El Corasonsito te bendice con<//><b>[$amount] Donante[s?$amount]</b> de Khron!
				yield $amount Donante
			end
		end
